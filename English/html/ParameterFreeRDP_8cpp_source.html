61 bool bUseMultimon = set.value("FreeRDP/UseMultimon", freerdp_settings_get_bool(
62 m_pSettings, FreeRDP_UseMultimon)).toBool();
63 freerdp_settings_set_bool(m_pSettings, FreeRDP_UseMultimon, bUseMultimon);
64
65 SetDesktopSizes(set.value("FreeRDP/DesktopSizes",
66 GetDesktopSizes()).toStringList());
67 SetReconnectInterval(set.value("FreeRDP/ReconnectionInterval",
68 GetReconnectInterval()).toInt());
69
70 SetShowVerifyDiaglog(set.value("FreeRDP/ShowVerifyDiaglog",
71 GetShowVerifyDiaglog()).toBool());
72
73 SetRedirectionPrinter(set.value("FreeRDP/Redirection/Printer",
74 GetRedirectionPrinter()).toBool());
75 SetRedirectionSound(static_cast<RedirecionSoundType>(set.value("FreeRDP/Redirection/Sound",
76 static_cast<int>(GetRedirectionSound())).toInt()));
77 SetRedirectionSoundParameters(set.value("FreeRDP/Redirection/Sound/Parameters",
78 GetRedirectionSoundParameters()).toString());
79 SetRedirectionMicrophone(set.value("FreeRDP/Redirection/Microphone",

81 SetRedirectionMicrophoneParameters(set.value("FreeRDP/Redirection/Microphone/Parameters",
82 GetRedirectionMicrophoneParameters()).toString());
83
84 QStringList lstDrive;
85 int count = set.value("FreeRDP/Redirection/Drive/count", 0).toInt();
86 for(int i = 0; i < count; i++)
87 {
88 QString drive = set.value("FreeRDP/Redirection/Drive/" + QString::number(i), "").toString();
89 if(!drive.isEmpty())
90 lstDrive.append(drive);
91 }
92 if(!lstDrive.isEmpty())
93 SetRedirectionDrives(lstDrive);
94

149void CParameterFreeRDP::SetReconnectInterval(UINT newReconnectInterval)
150{
151 Q_ASSERT(newReconnectInterval >= 0);
152 if(newReconnectInterval)
153 freerdp_settings_set_bool(m_pSettings,
154 FreeRDP_AutoReconnectionEnabled, true);
155 else
156 freerdp_settings_set_bool(m_pSettings,
157 FreeRDP_AutoReconnectionEnabled, false);
158
159 if (m_nReconnectInterval == newReconnectInterval)
160 return;
161
162 m_nReconnectInterval = newReconnectInterval;
163 freerdp_settings_set_uint32(m_pSettings,
164 FreeRDP_AutoReconnectMaxRetries,
165 m_nReconnectInterval);
166
167 SetModified(true);
168 emit sigReconnectIntervalChanged();
169}
